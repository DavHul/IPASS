64 uint8_t lezen_secondes(){
65 { hwlib::i2c_write_transaction wtrans = ((hwlib::i2c_bus*)(&bus))->write(adres);
66 wtrans.write(adres_secondes);}
67 { hwlib::i2c_read_transaction rtrans = ((hwlib::i2c_bus*)(&bus))->read(adres);
68 rtrans.read(secondes);}
69 if ((secondes >> 7 & 0x01)==1){ //als oscilator uitstaat kan je nog steeds de secondes aflezen door het meest linker bit op 0 te zetten
70 secondes = secondes ^ 0x80; //meest linker bit op 0 te zetten
71 }
72 return bcd2dec(secondes);
73 }
